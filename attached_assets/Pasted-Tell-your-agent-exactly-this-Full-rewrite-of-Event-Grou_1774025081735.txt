**Step 3 — build chains with real matched markets:**
```javascript
const buildChain = (chain, markets) => {
  const sideA = markets.filter(m => {
    const q = m.question.toLowerCase()
    return chain.sideA.keywords.some(kw => q.includes(kw))
  }).slice(0, 6)

  const sideB = markets.filter(m => {
    const q = m.question.toLowerCase()
    return chain.sideB.keywords.some(kw => q.includes(kw))
  }).slice(0, 6)

  // BOTH sides must have real markets — this is the whole point
  if (sideA.length === 0 || sideB.length === 0) return null
  
  // Need at least 2 markets total across both sides
  if (sideA.length + sideB.length < 3) return null

  const allMarkets = [...sideA, ...sideB]
  const totalVol = allMarkets.reduce((s, m) => 
    s + parseFloat(m.volume24hr || 0), 0)

  // Check if any market moved significantly in 24h
  const movedMarkets = allMarkets.filter(m => 
    Math.abs(parseFloat(m.priceChange24h || 0)) > 0.03
  )

  return {
    ...chain,
    sideA,
    sideB,
    totalVol,
    hasMovement: movedMarkets.length > 0,
    movedCount: movedMarkets.length
  }
}

const chains = CAUSAL_CHAINS
  .map(c => buildChain(c, markets))
  .filter(Boolean)
  .sort((a, b) => b.totalVol - a.totalVol) // highest vol chains first
```
