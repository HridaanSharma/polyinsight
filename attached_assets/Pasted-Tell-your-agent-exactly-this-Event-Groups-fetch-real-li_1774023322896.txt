**Event Groups — fetch real live data, build causal chains from it:**

```javascript
// Step 1: Fetch real markets from Gamma API
const res = await fetch('/proxy/gamma/markets?active=true&closed=false&limit=500&order=volume24hr&ascending=false')
const allMarkets = await res.json()

// Step 2: Filter to meaningful markets only
const meaningful = allMarkets.filter(m => {
  try {
    const yes = parseFloat(JSON.parse(m.outcomePrices)[0])
    return yes > 0.05 && yes < 0.95 &&
      parseFloat(m.volume24hr) > 10000 &&
      m.active === true &&
      m.closed === false
  } catch { return false }
})

// Step 3: Define causal chain themes with keywords
const CAUSAL_CHAINS = [
  {
    theme: "Iran Military Escalation",
    description: "If US invades Iran → oil spikes → regime collapses",
    keywords: ["iran", "invade iran", "iranian regime", "crude oil", "hormuz", "kharg"]
  },
  {
    theme: "Fed & US Economy",
    description: "Inflation drives Fed decisions which drive market prices",
    keywords: ["fed ", "federal reserve", "interest rate", "inflation", "recession", "s&p 500", "unemployment"]
  },
  {
    theme: "Trump Foreign Policy",
    description: "Trump actions abroad affect trade, alliances and economies",
    keywords: ["trump", "tariff", "china", "taiwan", "trade war", "sanctions"]
  },
  {
    theme: "2026 US Midterms",
    description: "Who controls Congress determines what policy passes",
    keywords: ["midterm", "senate", "house of representatives", "democrat", "republican", "governor", "congress"]
  },
  {
    theme: "AI & Tech Race",
    description: "AI milestones and company moves are interconnected",
    keywords: ["openai", "anthropic", "google deepmind", "agi", "gpt", "ai model", "nvidia"]
  },
  {
    theme: "Russia-Ukraine War",
    description: "Ceasefire and territorial outcomes are deeply linked",
    keywords: ["ukraine", "russia", "putin", "zelensky", "ceasefire", "nato", "crimea"]
  },
  {
    theme: "Crypto Market",
    description: "Bitcoin ETF, regulation and price targets move together",
    keywords: ["bitcoin etf", "sec crypto", "crypto regulation", "bitcoin above", "ethereum above"]
  }
]

// Step 4: Match real markets to each chain
const chains = CAUSAL_CHAINS.map(chain => {
  const matched = meaningful.filter(m => {
    const q = m.question.toLowerCase()
    const eventTitle = (m.groupItemTitle || m.question || '').toLowerCase()
    return chain.keywords.some(kw => q.includes(kw) || eventTitle.includes(kw))
  })

  // Deduplicate by conditionId
  const unique = matched.filter((m, i, arr) => 
    arr.findIndex(x => x.conditionId === m.conditionId) === i
  )

  // Need at least 3 real markets to show the chain
  if (unique.length < 3) return null

  return {
    ...chain,
    markets: unique.slice(0, 8), // max 8 per chain
    totalVolume: unique.reduce((sum, m) => sum + parseFloat(m.volume24hr || 0), 0)
  }
}).filter(Boolean)
```

```javascript
            </div>
          `
        }).join('')}
      </div>
    </div>
  `
})
```

**Volume Spikes — real data, filtered properly:**

```javascript
const spikes = allMarkets
  .filter(m => {
    const q = m.question.toLowerCase()
    const prob = parseFloat(JSON.parse(m.outcomePrices)[0])
    const vol24 = parseFloat(m.volume24hr || 0)
    const volTotal = parseFloat(m.volumeClob || m.volume || 1)
    const daysOld = (Date.now() - new Date(m.startDate || m.createdAt)) / 86400000
    
    // Skip noise
    const isNoise = [
      ' vs ', ' vs. ', ' @ ', 'tweets', 'o/u ',
      'win on 2026', 'win on 2025', 'drivers champion',
      'eurovision', 'ncaa tournament'
    ].some(kw => q.includes(kw))
    if (isNoise) return false

    // Must be genuinely uncertain
    if (prob < 0.08 || prob > 0.88) return false

    // Must have real volume today
    if (vol24 < 200000) return false

    // Spike score = today vs historical daily average
    const avgDaily = volTotal / Math.max(daysOld, 1)
    const spike = vol24 / avgDaily
    m._spike = spike
    m._avgDaily = avgDaily

    return spike >= 4 && daysOld > 7 // must be old enough to have a baseline
  })
  .sort((a, b) => b._spike - a._spike)
  .slice(0, 15)
```

```javascript
// Gamma API returns a 'url' field on most markets — use it directly
const getTradeUrl = (market) => {

  if (market.slug) return `https://polymarket.com/event/${market.slug}`
  return `https://polymarket.com`
}
```
