  const eventSlug = event.slug  // this is the real URL slug
  if (event.markets) {
    event.markets.forEach(m => {
      // Real URL format: https://polymarket.com/event/[eventSlug]
      urlMap[m.conditionId] = `https://polymarket.com/event/${eventSlug}`
    })
  }
})

// Step 3: When building any trade button, look up URL from map
const getTradeUrl = (market) => {
  if (urlMap[market.conditionId]) return urlMap[market.conditionId]
  // fallback: use event.url field if it exists on the market object
  if (market.event?.slug) return `https://polymarket.com/event/${market.event.slug}`
  return `https://polymarket.com`
}
