**Fix 2 — Volume Spikes: scan ALL 500 markets purely mathematically:**

```javascript
const res = await fetch('/proxy/gamma/markets?active=true&closed=false&limit=500&order=volume24hr&ascending=false')
const all = await res.json()

const spikes = all
  .filter(m => {
    try {
      const vol24 = parseFloat(m.volume24hr || 0)
      const volTotal = parseFloat(m.volumeClob || m.volume || 0)
      const daysOld = (Date.now() - new Date(m.startDate || m.createdAt)) / 86400000
      const yes = parseFloat(JSON.parse(m.outcomePrices)[0])

      // Pure math filters only
      if (daysOld < 21) return false        // needs baseline
      if (vol24 < 100000) return false      // real money today
      if (volTotal < 100000) return false   // enough history
      if (yes < 0.06 || yes > 0.94) return false // genuinely uncertain

      const avgDaily = volTotal / daysOld
      if (avgDaily < 1000) return false     // skip markets with tiny history

      const spike = vol24 / avgDaily
      m._spike = spike
      m._avgDaily = avgDaily
      m._daysOld = Math.round(daysOld)

      return spike >= 3
    } catch { return false }
  })
  .sort((a, b) => b._spike - a._spike)
  .slice(0, 20)
```

**Fix 3 — Spread Scanner: scan ALL 500 markets using Gamma bestBid/bestAsk but fetch live CLOB for top candidates:**

```javascript
// Phase 1: use Gamma data to find candidates fast
const res = await fetch('/proxy/gamma/markets?active=true&closed=false&limit=500&order=volume24hr&ascending=false')
const all = await res.json()

// Find candidates with Gamma data first
const candidates = all.filter(m => {
  const bid = parseFloat(m.bestBid || 0)
  const ask = parseFloat(m.bestAsk || 0)
  const vol24 = parseFloat(m.volume24hr || 0)
  if (!bid || !ask) return false
  if (bid < 0.05 || ask > 0.95) return false
  if (vol24 < 1000) return false
  const spread = ask - bid
  return spread > 0.01
}).sort((a,b) => {
  const spreadA = parseFloat(a.bestAsk) - parseFloat(a.bestBid)
  const spreadB = parseFloat(b.bestAsk) - parseFloat(b.bestBid)
  return spreadB - spreadA
}).slice(0, 50) // top 50 candidates

// Phase 2: fetch live CLOB for these 50 in parallel
const liveSpread = async (market) => {
  try {
    const tokenId = JSON.parse(market.clobTokenIds)[0]
    const res = await fetch(`/proxy/clob/book?token_id=${tokenId}`)
    const book = await res.json()

    if (!book.bids?.length || !book.asks?.length) return null

    const bestBid = parseFloat(book.bids[0].price)
    const bestAsk = parseFloat(book.asks[0].price)
    const spread = bestAsk - bestBid

    if (bestBid < 0.05 || bestAsk > 0.95) return null
    if (spread < 0.01) return null

    return {
      question: market.question,
      conditionId: market.conditionId,
      slug: market.slug,
      url: market.url,
      bestBid,
      bestAsk,
      spread,
      spreadCents: Math.round(spread * 100),
      volume24hr: parseFloat(market.volume24hr)
    }
  } catch { return null }
}

const results = await Promise.all(candidates.map(liveSpread))
const validSpreads = results
  .filter(Boolean)
  .sort((a,b) => b.spread - a.spread)
  .slice(0, 30)
```

This scans all 500 markets for candidates using fast Gamma data, then verifies the top 50 with live CLOB data. You get real spreads across the entire platform, not just pre-filtered categories.
