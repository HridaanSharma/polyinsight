    return chain.keywordsGroupA.some(kw => q.includes(kw))

    return chain.keywordsGroupB.some(kw => q.includes(kw))

  )

  if (combined.length < 3) return null

    ...chain,
    groupA,
    groupB,
    markets: combined
  }
}

**Problem 2 — Trade button 404**

The slug is wrong. Tell your agent:

```javascript
// Gamma API has a field called 'url' on each market object
// Log one market to console to see all available fields:
console.log(JSON.stringify(allMarkets[0], null, 2))

// Then use whichever URL field exists:
const getTradeUrl = (market) => {
  // Try these in order — first one that exists wins
  if (market.url && market.url.includes('polymarket.com')) 
    return market.url
  if (market.conditionId) 
    return `https://polymarket.com/event/${market.slug}?tid=${market.conditionId}`
  return `https://polymarket.com/event/${market.slug}`
}

// Add this debug line temporarily to see what URL fields exist:
console.log('URL fields:', {
  url: allMarkets[0].url,
  slug: allMarkets[0].slug,  
  conditionId: allMarkets[0].conditionId,
  groupItemTitle: allMarkets[0].groupItemTitle
})
```

The correct Polymarket URL format is actually:
```
https://polymarket.com/event/[event-slug]
```
NOT the market slug — the EVENT slug. Each market has a `marketType` and belongs to an event. Tell the agent to use `market.events?.[0]?.slug` if available, otherwise fall back to `market.slug`.
